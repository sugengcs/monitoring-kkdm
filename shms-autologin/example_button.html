Auto Login SHMS - Contoh Implementasi

1. Button dengan JavaScript (Recommended)

Ketika user klik tombol, JavaScript akan request token ke server dan redirect ke SHMS.

2. Link Direct dengan Token

Jika Anda sudah memiliki token, gunakan link direct ini:

<a href="https://shms.risen.id/autologin.php?token=YOUR_TOKEN">
    Login ke SHMS
</a>

3. Implementasi di PHP Website

Contoh kode PHP untuk generate token dan redirect:

<?php
// Di website utama Anda
require_once 'path/to/generate_token.php';

// Generate token untuk user tertentu
$username = 'admin'; // atau user_id
$result = generateTokenByUsername($username);

if ($result['success']) {
    // Redirect ke SHMS dengan token
    header('Location: ' . $result['autologin_url']);
    exit;
} else {
    echo "Error: " . $result['message'];
}
?>

4. Implementasi dengan AJAX

Contoh implementasi menggunakan AJAX untuk generate token:

// JavaScript
function generateTokenWithAjax(username) {
    fetch('https://website-utama.com/generate_token.php', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            username: username
        })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            window.location.href = data.autologin_url;
        } else {
            alert('Error: ' + data.message);
        }
    })
    .catch(error => {
        alert('Error: ' + error.message);
    });
}

5. Implementasi di React/Vue/Angular

Contoh implementasi di React:

// React Component
import { useState } from 'react';

function AutoLoginButton({ username }) {
    const [loading, setLoading] = useState(false);
    
    const handleAutoLogin = async () => {
        setLoading(true);
        try {
            const response = await fetch('https://website-utama.com/generate_token.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ username })
            });
            const data = await response.json();
            
            if (data.success) {
                window.location.href = data.autologin_url;
            } else {
                alert('Error: ' + data.message);
            }
        } catch (error) {
            alert('Error: ' + error.message);
        } finally {
            setLoading(false);
        }
    };
    
    return (
        <button onClick={handleAutoLogin} disabled={loading}>
            {loading ? 'Loading...' : 'Login ke SHMS'}
        </button>
    );
}
